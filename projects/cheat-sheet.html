[text] 

// JavaScript

// Semicolons
var myNumber = 0;
myNumber += 2;
console.log( myNumber > 0 )
=> true

// Conditional Parentheses
if( myBoolean )
{
  console.log( "My boolean is true!" )
}
else if( !myBoolean )
{
  console.log( "My boolean is false!" )
}
else
{
  console.log( "Uhh...something went wrong." )
}

// Function Parentheses
var myNumber = 100248;
var myString = myNumber.toString()
function printMyString()
{
  console.log( myString )
}
=> "100248"

// Explanation

// Semicolons
Only the first two expressions here need
semicolons: each acts as a separator in JS, so
the last expression doesn't need one. However,
leaving out semicolons makes your code prone to
errors when you update ( like if we added another
expression after these three ), so I like to
include them even where they aren't strictly
needed.


// Conditional Parentheses
Here, the parentheses wrap our entire
conditional, this allows us to place multiple
conditionals in a row ( with && and || for
example ) and still make it visually obvious
where the conditionals end and the code block
begins.


// Function Parentheses
A function call or declaration must always be
followed by paretheses even if the function
accepts no parameters. This communicates to
JavaScript that we do not want to try to pass any
parameters, and, like Conditional Parentheses,
makes it visually obvious that we're making a
function call.

[section 2]

[text] 

// JavaScript

// toString()
var myNumber = 4.553;
console.log( myNumber.toString() )
=> "4.533"


// hasOwnProperty()
var myObject = {
  prop: "A property",
};
console.log( myObject.hasOwnProperty( "prop" ) )
=> true


// Explanation

// toString()
Any JavaScript object can be converted to a
string. In fact, JavaScript will do this
automatically if, for example, you try to compare
a string with an object that is not a string.

// hasOwnProperty()
This function is very powerful: it only checks
those properties NOT inherited from prototypes.
For example this allows you to iterate through an
object you've created in the same way you would
iterate through a hash in Ruby.

[section 3]

[text] 

// JavaScript

// Iterate over an object
var myObject = {
  propString: "A property",
  propNum: 256,
	propArray: [ "This", "is", "an", "array", "." ],
};
for( var key in myObject )
{
	if( myObject.hasOwnProperty( key ) )
	{
    console.log( myObject[ key ] );
  }
}
=> "A property"
=> 256
=> [ "This", "is", "an", "array", "." ]


// Explanation

// Iterate over an object
Here, we can use 'hasOwnProperty' to only check
those values that match the properties we
assigned to 'myObject'. You can also use a nested
'for' loop to iterate over objects stored within
the properties of another object.

// JavaScript

// STRING methods
myString.concat( anotherString, aThirdString );



myString.split( otherString );





myString.slice( startIndex, endIndex );



myString.trim();



// ARRAY methods
myArray.concat( myObject, otherObject, ... );



myArray.push( myObject, otherObject, ... );


myArray.join( myString );




myArray.unshift( myObject, otherObject, ... );



myArray.splice( index, numToDelete,
  myObject, otherObject, ... );




myArray.reverse()


// Explanation

// STRING methods
returns a new string consisting of 'myString'
followed by 'anotherString', 'aThirdString', and
however many other parameters you need

returns a new array where each element is a
segment of 'myString' split based on
'otherString'; if 'otherString' is empty, the new
array will contain every character of 'myString'
including spaces

returns a new string starting from the character
index 'startIndex', up to and including the
character at 'endIndex'

returns a new string from 'myString' with extra
spaces removed: this includes spaces at the
beginning and end of 'myString'

// ARRAY methods
returns a new array containing 'myArray' with
'myObject' and 'otherObject' added as new
elements to the end of the array

adds 'myObject' and 'otherObject' to the end of
'myArray' and returns the new length of 'myArray'

returns a new string where each element of
'myArray' is separated by 'myString'; if no
parameter is passed to 'join()', each element is
separated by a comma.

adds 'myObject', 'otherObject', and so on to the
beginning of 'myArray', moving all other elements
up to the next index for each new element.

adds 'myObject' and 'otherObject' to 'myArray'
starting at index 'index', and deleting
'numToDelete' number of elements starting at
'index'. Can be used to remove elements from an
array by not passing parameters 'myObject', etc.

reorders the elements of 'myArray' in reverse
order based on index number
